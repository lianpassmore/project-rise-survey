    <script>

        let currentSessionId = null;


        // Function to create session in Supabase via API

        async function createSession() {

            try {

                const response = await fetch('/api/create-session', {

                    method: 'POST',

                    headers: {

                        'Content-Type': 'application/json',

                    }

                });


                if (!response.ok) {

                    throw new Error('Failed to create session');

                }


                const data = await response.json();

                return data.sessionId;

            } catch (error) {

                console.error('Error creating session:', error);

                // Fallback to local session ID if API fails

                return 'RISE_' + Math.random().toString(36).substr(2, 9) + '_' + Date.now();

            }

        }


        // Initialize session on page load

        async function initializeSession() {

            try {

                // Check if we already have a session ID

                let sessionId = localStorage.getItem('trurivu_session_id');

                

                if (!sessionId) {

                    // Create new session

                    sessionId = await createSession();

                    localStorage.setItem('trurivu_session_id', sessionId);

                }


                currentSessionId = sessionId;

                

                // Update UI

                document.getElementById('session-display').textContent = sessionId.substring(0, 20) + '...';

                document.getElementById('session-info').style.display = 'block';

                document.getElementById('loading').style.display = 'none';

                document.getElementById('form-container').style.display = 'block';

                

                // Add session ID to URL for form to pick up

                const url = new URL(window.location);

                url.searchParams.set('session_id', sessionId);

                window.history.replaceState({}, '', url);

                

            } catch (error) {

                console.error('Error initializing session:', error);

                document.getElementById('loading').textContent = 'Error setting up session. Please refresh the page.';

            }

        }


        // Listen for form completion

        window.addEventListener('message', function(event) {

            if (event.data && event.data.type === 'fillout:form_completed') {

                // Redirect to interview page with session ID

                if (currentSessionId) {

                    window.location.href = `interview.html?session=${currentSessionId}`;

                }

            }

        });


        // Initialize when page loads

        window.addEventListener('load', initializeSession);

    </script>
